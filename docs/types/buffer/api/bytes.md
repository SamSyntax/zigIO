- #### Example

    ```zig
    const chars = @import("io").utils.chars;
    const buffer = @import("io").types.buffer;
    ```

    ```zig
    var buf = chars.make(64, null);
    var str = buffer(&buf);

    _ = str.bytes(); // 👉 0

    try str.append("Hello 🌍!");

    _ = str.bytes(); // 👉 11
    ```
